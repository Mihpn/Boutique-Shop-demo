===============================================================================
  PHASE 2: KUBERNETES CLUSTER PREREQUISITES
===============================================================================

# --- 2.1: Create Namespace ---
kubectl create namespace online-boutique
kubectl create namespace online-boutique --dry-run=client -o yaml | kubectl apply -f -

# --- 2.2: Create Docker Hub Pull Secret ---
kubectl create secret docker-registry dockerhub-key \
  --docker-server=https://index.docker.io/v1/ \
  --docker-username=kihpn1711 \
  --docker-password=<YOUR_DOCKERHUB_ACCESS_TOKEN> \
  --namespace=online-boutique

# --- 2.3: Create Gemini AI API Key Secret ---
kubectl create secret generic gemini-api-key \
  --from-literal=GEMINI_API_KEY="<YOUR_GEMINI_API_KEY>" \
  --namespace=online-boutique

# --- 2.4: Verify Secrets ---
kubectl get secrets -n online-boutique


===============================================================================
  PHASE 3: INSTALL ARGOCD ON KUBERNETES
===============================================================================

# --- 3.1: Add Helm Repository ---
helm repo add argo https://argoproj.github.io/argo-helm
helm repo update

# --- 3.2: Copy ArgoCD values file to master node (from Windows) ---
scp d:\DevOps\Projects\demo\microservices-demo\cicd\argocd-values.yaml \
  devops@<MASTER_NODE_IP>:~/argocd/values.yaml

# --- 3.3: Install ArgoCD via Helm ---
helm install argocd argo/argo-cd \
  --namespace argocd \
  --create-namespace \
  -f ~/argocd/values.yaml \
  --wait --timeout 5m

# --- 3.4: Retrieve Admin Password ---
kubectl get secret argocd-initial-admin-secret -n argocd \
  -o jsonpath="{.data.password}" | base64 -d && echo

# --- 3.5: Verify ArgoCD Installation ---
kubectl get pods -n argocd
kubectl get svc -n argocd
kubectl get ingress -n argocd

# --- 3.6: Access ArgoCD UI ---
#   URL   : http://argocd.kihpn.local
#   Login : admin / <password from step 3.4>

  PHASE 4: DEPLOY APPLICATION VIA ARGOCD
===============================================================================

# --- 4.1: Apply ArgoCD Application Manifest ---
kubectl apply -f cicd/argocd-application.yaml

# --- 4.2: Verify Application Status ---
kubectl get applications -n argocd

# --- 4.3: Watch Pods Starting (5-Group Startup Order) ---
kubectl get pods -n online-boutique -w

# Expected startup order:
#   Group 1: redis-cart                    (Data Layer)
#   Group 2: adservice, currencyservice,   (Independent Backends)
#            emailservice, paymentservice,
#            shippingservice, productcatalogservice
#   Group 3: cartservice,                  (Dependent Backends)
#            recommendationservice,
#            checkoutservice
#   Group 4: shoppingassistantservice,     (AI & Frontend)
#            frontend
#   Group 5: ingress                       (Traffic Entry)

# --- 4.4: Verify All Pods Running ---
kubectl get pods -n online-boutique
# Expected: 12 pods, all showing 1/1 Running

# --- 4.5: Verify Services ---
kubectl get svc -n online-boutique

# --- 4.6: Verify Ingress ---
kubectl get ingress -n online-boutique

===============================================================================
  PHASE 7: UPGRADE & ROLLBACK PROCEDURES
===============================================================================

# --- 7.1: Standard Upgrade (GitOps) ---
# 1. Edit values.yaml → change image tag
# 2. git add . && git commit -m "chore: bump frontend to v3"
# 3. git push origin main
# 4. ArgoCD auto-syncs within 3 minutes

# --- 7.2: Manual Sync (Immediate) ---
kubectl patch application online-boutique -n argocd \
  --type merge -p '{"operation":{"sync":{"revision":"HEAD"}}}'

#   Or via ArgoCD CLI:
argocd app sync online-boutique

# --- 7.3: Emergency Rollback ---
argocd app rollback online-boutique 1

# --- 7.4: View Sync History ---
argocd app history online-boutique
